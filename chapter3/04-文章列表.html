export const getArticleList = (params) => {
  return async (dispatch) => {
    const res = await request({
      url: '/mp/articles',
      params
    })
    dispatch({
      type: GET_ARTICLE_LIST,
      payload: res.data.data
    })
  }
}

# 提供action的类型

export const GET_ARTICLE_LIST = 'GET_ARTICLE_LIST'

# 在reducers中处理action

const initValue = {
  channels: [],
  articles: {}
}
export default function article(state = initValue, action) {
  if (action.type === GET_CHANNEL_LIST) {
    return {
      ...state,
      channels: action.payload
    }
  }
  if (action.type === GET_ARTICLE_LIST) {
    return {
      ...state,
      articles: action.payload
    }
  }
  return state
}

# 状态处理列

# 目标

状态列的格式转换

将状态转成 视图上的tag。

Tag组件： 文档 (opens new window)

# 背景

后端的编号对应的状态是：

-1： 全部， 0： 草稿 ，1：待审核， 2：审核通过，3：审核失败

# 思路

额外补充定义常量数组，

数据提取，用数组来渲染

const STATUS = [
  { id: -1, title: '全部', color: 'magenta' },
  { id: 0, title: '草稿', color: 'red' },
  { id: 1, title: '待审核', color: 'volcano' },
  { id: 2, title: '审核通过', color: 'lime' },
  { id: 3, title: '审核失败', color: 'gold' }
]

// 表单
<Form.Item label="状态" name="status">
  <Radio.Group>
    {STATUS.map((item) => (
      <Radio key={item.id} value={item.id}>
        {item.title}
      </Radio>
    ))}
  </Radio.Group>
</Form.Item>

表格数据渲染


    
{
  title: '状态',
  dataIndex: 'status',
  render: (status) => {
    const obj = STATUS.find((item) => item.id === status)
    return <Tag color={obj.color}>{obj.title}</Tag>
  }
},

# 内容管理-删除文章（弹窗确认）

# 目标

能够实现点击删除按钮时弹窗确认

# 步骤

  1. 给删除文章按钮绑定点击事件
  2. 弹出确认窗口，询问用户是否确定删除文章

# 核心代码

  • 提供一个删除的action
  • actions/article.js
export const delArticle = (id) => {
  return async () => {
    await request({
      method: 'delete',
      url: `/mp/articles/${id}`
    })
  }
}
  • 在组件中，弹窗提醒
{
  title: '操作',
  dataIndex: 'id',
  render(id) {
    return (
      <Space>
        <Button shape="circle" type="primary" icon={<EditOutlined />} />
        <Popconfirm title="确定要删除该文章吗？" onConfirm={() => del(id)}>
          <Button shape="circle" type="danger" icon={<DeleteOutlined />} />
        </Popconfirm>
      </Space>
    )
  }
}
  • 提供del方法，进行删除
const del = async (id) => {
  await dispatch(delArticle(id))
  message.success('删除成功')
  // 重新发送请求
  await dispatch(getArticleList(params.current))
  
}
